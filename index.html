async function fetchOfficialJobs(query, elementId) { try { // हमने यहाँ 'site' फ़िल्टर और 'official' कीवर्ड्स जोड़ दिए हैं // ताकि डेटा विश्वसनीय जगहों (जैसे .gov.in या प्रमुख जॉब पोर्टल) से आए const officialQuery = `${query} (site:gov.in OR "official notification" OR "vacancy") -blog -news`; const url = `https://news.google.com/rss/search?q=${encodeURIComponent(officialQuery)}&hl=hi&gl=IN&ceid=IN:hi`; const res = await fetch(`https://api.rss2json.com/v1/api.json?rss_url=${encodeURIComponent(url)}`); const data = await res.json(); // भीड़ कम करने के लिए सिर्फ टॉप 3 सबसे सटीक रिजल्ट्स const cleanItems = data.items.slice(0, 3); if (cleanItems.length === 0) { document.getElementById(elementId).innerHTML = "अभी कोई आधिकारिक सूचना नहीं है।"; return; } document.getElementById(elementId).innerHTML = cleanItems.map(item => `
OFFICIAL SOURCE
${item.title.split(' - ')[0]}
नोटिफिकेशन देखें →
`).join(''); } catch(e) { document.getElementById(elementId).innerHTML = "सर्वर से डेटा नहीं मिल रहा।"; } } // इन सर्च कीवर्ड्स को मैंने और भी 'टाइट' कर दिया है fetchOfficialJobs("ITI Fitter Permanent Recruitment 2026", "present-list"); fetchOfficialJobs("Upcoming Railway Technician recruitment 2026", "upcoming-list"); fetchOfficialJobs("Railway NCL Apprentice Merit List 2026", "apprentice-list");
