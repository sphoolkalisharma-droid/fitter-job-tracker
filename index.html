async function fetchData(fileName, elementId) { try { const response = await fetch(`./${fileName}?nocache=${Date.now()}`); if (!response.ok) throw new Error("File Missing"); const data = await response.json(); let html = data.slice(0, 10).map(item => `
${item.title || item.question || "New Entry"}
क्लिक करें →
`).join(''); document.getElementById(elementId).innerHTML = html; } catch (error) { // अगर फाइल खराब है, तो ये बताएगा document.getElementById(elementId).innerHTML = "फाइल पढ़ने में गलती है! (JSON Error)"; } }
