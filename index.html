async function loadData() { // 1. Jobs लोड करने की कोशिश try { const jobRes = await fetch('jobs.json'); // पक्का करें कि नाम jobs.json ही है if (!jobRes.ok) throw new Error('jobs.json नहीं मिली'); const jobs = await jobRes.json(); document.getElementById('present-list').innerHTML = jobs.map(j => `
${j.title || j.department || "Job Details"}
` ).join(''); } catch (e) { document.getElementById('present-list').innerHTML = `
चेक करें: jobs.json फाइल खाली तो नहीं?
`; } // 2. Questions लोड करने की कोशिश try { const qRes = await fetch('question.json'); // पक्का करें कि नाम question.json ही है if (!qRes.ok) throw new Error('question.json नहीं मिली'); const qs = await qRes.json(); document.getElementById('question-list').innerHTML = qs.map(q => `
${q.question || q.text || "Question Details"}
` ).join(''); } catch (e) { document.getElementById('question-list').innerHTML = `
चेक करें: question.json फाइल खाली तो नहीं?
`; } } loadData(); // इसे अंत में जरूर लिखें
